import React, { useState } from 'react';
import { Question, Answer } from '../QandA';
import { AboutContainer } from './FAQ.styles';

const FAQ = (props) => {
  const [open, setOpen] = useState(false);
  const handleOpen = () => {
    if (open === true) {
      setOpen(false);
    } else {
      setOpen(true);
    }
  };
  return (
    <AboutContainer>
      {props.data.map((rows) => (
        <>
          <Question
            question={rows.question}
            isOpen={open}
            onClick={handleOpen}
            answerId={rows.id}
          />
          <Answer
            answer={rows.answer}
            isOpen={open}
            onToggle={handleOpen}
            answerId={rows.id}
          />
        </>
      ))}
    </AboutContainer>
  );
};

export { FAQ };
